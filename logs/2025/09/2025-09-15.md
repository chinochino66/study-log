# STEP2：スコア機能をつける

```python
import random

class KnowledgeCard:
    def __init__(self, question, answer):
        self.question = question
        self.answer = answer

    def quiz(self) -> bool:
        print("Q:", self.question)
        user = input("答えは？ > ")
        # 入力と正解を両方トリム＆小文字化して比較（表記ゆれ吸収）
        if user.strip().lower() == self.answer.strip().lower():
            print("✅ 正解！\n")
            return True
        else:
            print("❌ 不正解。正解は:", self.answer, "\n")
            return False

def run_quiz(cards: list[KnowledgeCard]) -> None:
    quiz_cards = cards[:]          # 元を壊さない
    random.shuffle(quiz_cards)     # ランダム出題（不要ならこの行を消す）

    correct = 0                    # ← ループの外で初期化！
    total = len(quiz_cards)

    for card in quiz_cards:
        if card.quiz():            # ← 戻り値(True/False)で加算
            correct += 1

    print(f"結果：{total}問中 {correct}問正解！")

cards = [
    KnowledgeCard("Pythonで関数を定義するキーワードは？", "def"),
    KnowledgeCard("文字列の型は？", "str"),
    KnowledgeCard("リストに要素を追加するメソッドは？", "append"),
]
```
